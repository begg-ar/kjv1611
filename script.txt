import fontforge
font = fontforge.open("KJV1611.sfd")
import spaces

font.encoding = "UnicodeBMP"

for k in spaces.Spaces.keys():
    font.createChar(k)

for k, v in spaces.Spaces.items():
    font[k].width = v

font.selection.select(("unicode","ranges"),0xE000,0xE0FF)
font.clear()
font.selection.all()
font.unlinkReferences()
font.removeOverlap()
font.mergeFeature("features.fea")
font.generate("KJV1611.otf", flags=("opentype", "old-kern", "no-hints", "no-flex", "winkern"))
